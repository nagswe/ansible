vpc cration

---
- name: Create VPC on AWS
  hosts: localhost
  gather_facts: no
  tasks:
    - name: Create a VPC
      amazon.aws.ec2_vpc:
        cidr_block: "10.0.0.0/16"
        state: present
        region: "us-west-2"
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"
        resource_tags:
          Name: "MyVPC"
      register: vpc

    - name: Create a subnet in the VPC
      amazon.aws.ec2_vpc_subnet:
        vpc_id: "{{ vpc.vpc.id }}"
        cidr: "10.0.1.0/24"
        region: "us-west-2"
        state: present
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"
        resource_tags:
          Name: "MySubnet"
      register: subnet

    - name: Create an Internet Gateway
      amazon.aws.ec2_vpc_igw:
        vpc_id: "{{ vpc.vpc.id }}"
        region: "us-west-2"
        state: present
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"
        resource_tags:
          Name: "MyInternetGateway"
      register: igw

    - name: Attach the Internet Gateway to the VPC
      amazon.aws.ec2_vpc_igw_attachment:
        vpc_id: "{{ vpc.vpc.id }}"
        internet_gateway_id: "{{ igw.internet_gateway.id }}"
        region: "us-west-2"
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"

    - name: Create a Route Table for the VPC
      amazon.aws.ec2_vpc_route_table:
        vpc_id: "{{ vpc.vpc.id }}"
        region: "us-west-2"
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"
        resource_tags:
          Name: "MyRouteTable"
      register: route_table

    - name: Create a route to the Internet
      amazon.aws.ec2_vpc_route:
        route_table_id: "{{ route_table.route_table.id }}"
        destination_cidr_block: "0.0.0.0/0"
        gateway_id: "{{ igw.internet_gateway.id }}"
        region: "us-west-2"
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"

    - name: Associate the Route Table with the Subnet
      amazon.aws.ec2_vpc_subnet_route_table:
        subnet_id: "{{ subnet.subnet.id }}"
        route_table_id: "{{ route_table.route_table.id }}"
        region: "us-west-2"
        aws_access_key: "{{ aws_access_key }}"
        aws_secret_key: "{{ aws_secret_key }}"
